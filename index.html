    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const modal = document.getElementById('imageModal');

            const modalImg = document.getElementById('modalImage');

            const modalCaption = document.getElementById('modalCaption');

            const modalFilename = document.getElementById('modalFilename');

            const closeBtn = document.getElementById('closeModal');


            // Modal Handlers

            const closeModal = () => {

                modal.classList.add('hidden');

                modal.classList.remove('flex');

            };

            closeBtn.addEventListener('click', closeModal);

            modal.addEventListener('click', (e) => {

                if (e.target === modal) closeModal();

            });

            document.addEventListener('keydown', (e) => {

                if (e.key === 'Escape') closeModal();

            });


            // Target all primary list items in the menu cards

            const menuItems = document.querySelectorAll('.menu-card li');

            

            menuItems.forEach(item => {

                // Ignore sub-table add-ons (like protein add-on choices)

                if (item.closest('.grid-cols-2')) return;


                // Find the name container (usually the first span)

                const nameElement = item.querySelector('span');

                

                if (nameElement) {

                    const rawName = nameElement.textContent.trim();

                    

                    // Create clean filename: remove leading numbers (e.g. "1. "), remove special chars/spaces

                    const cleanName = rawName.replace(/^\d+\.\s*/, '')

                                             .replace(/[^a-zA-Z0-9]/g, '')

                                             .toLowerCase();

                    

                    if (!cleanName) return;


                    const imageName = cleanName + '.jpg';


                    // Add interactive UI styling

                    item.classList.add('cursor-pointer', 'group', 'relative', 'hover:bg-slate-800/60', 'rounded-lg', 'transition-all', 'duration-300', 'p-2', '-mx-2');

                    

                    // Add subtle camera icon on hover

                    if(!nameElement.innerHTML.includes('fa-camera')) {

                        nameElement.innerHTML += ' <i class="fas fa-camera text-slate-400 text-sm ml-2 opacity-0 group-hover:opacity-100 transition-opacity transform group-hover:scale-110"></i>';

                    }


                    // Attach click listener to open modal

                    item.addEventListener('click', (e) => {

                        e.stopPropagation();

                        

                        modalImg.src = imageName;

                        

                        // Fallback logic if image file doesn't exist yet

                        modalImg.onerror = () => {

                            modalImg.src = `https://via.placeholder.com/800x600/1e293b/f8fafc?text=${imageName}`;

                        };

                        

                        // Clean caption for display

                        modalCaption.textContent = rawName.replace(/^\d+\.\s*/, '');

                        // Inform you of the exact file name it is looking for

                        modalFilename.textContent = `Requires file: ${imageName}`;

                        

                        modal.classList.remove('hidden');

                        modal.classList.add('flex');

                    });

                }

            });

        });
